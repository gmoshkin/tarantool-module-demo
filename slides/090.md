#[tarantool::proc]
fn insert(space_name: String, values: Vec<(usize, String)>)
    -> Result<(), MyError>
{
    let mut space = Space::find(&space_name)
      .ok_or_else(|| MyError::SpaceNotFound(space_name))?;

    for (key, value) in values {
        space.insert(&(key, value))?;
    }
    Ok(())
}
